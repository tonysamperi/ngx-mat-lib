xmat-accordion

A simpler way to declare a mat-accordion: just remember xmat-accordion, title, header and optionally desc

EXPANDED = {{accordionOpened ? "true" : "false"}}

DISABLED = {{accordionDisabled ? "true" : "false"}}

ACCORDION TITLE ({{accordionDisabled ? "DISABLED" : "ENABLED"}})

MY ACCORDION CONTENT

 
                
<button (click)="accordionOpened = !accordionOpened">
    ACCORDION TOGGLE OPEN
</button>
<p>EXPANDED = {{accordionOpened ? "true" : "false"}}</p>
<button (click)="accordionDisabled = !accordionDisabled">
    ACCORDION TOGGLE DISABLED
</button>
<p>DISABLED = {{accordionDisabled ? "true" : "false"}}</p>
<xmat-accordion [(expanded)]="accordionOpened"
                [disabled]="accordionDisabled"
                color="primary">
    <div class="xmat-accordion-title">
        ACCORDION TITLE ({{accordionDisabled ? "DISABLED" : "ENABLED"}})
    </div>
    <div class="xmat-accordion-content">
        <div class="my-content">
            <h2>MY ACCORDION CONTENT</h2>
        </div>
    </div>
</xmat-accordion>
